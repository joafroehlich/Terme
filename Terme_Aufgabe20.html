5x - 7y - {3z + 3y + [-4x + 11z - (10y - z) - (-3x + 8z)]}  

= 5x - 7y - {3z + 3y + [-4x + 11z - 10y + z + 3x - 8z]} 

= 5x - 7y - {3z + 3y + [- x + 4z - 10y]} 

= 5x - 7y - {3z + 3y - x + 4z - 10y} 

= 5x - 7y - {-x + 7z - 7y} 

= 5x - 7y + x - 7z + 7y 

= 6x - 7z
